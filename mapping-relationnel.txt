from django.db import models

class Artist(models.Model):
    # Clé primaire
    id = models.BigAutoField(primary_key=True, verbose_name="ID de l'artiste")
    firstname = models.CharField(max_length=60, verbose_name="Prénom de l'artiste")
    lastname = models.CharField(max_length=60, verbose_name="Nom de l'artiste")

class Type(models.Model):
    # Clé primaire
    id = models.BigAutoField(primary_key=True, verbose_name="ID du type")
    type = models.CharField(max_length=60, verbose_name="Fonction de l'artiste (métier)")

class ArtistType(models.Model):
    # Clé primaire
    id = models.BigAutoField(primary_key=True, verbose_name="ID de la relation artiste-type")
    # Clé étrangère vers Artist
    artist = models.ForeignKey(Artist, on_delete=models.CASCADE, verbose_name="Artiste")
    # Clé étrangère vers Type
    type = models.ForeignKey(Type, on_delete=models.CASCADE, verbose_name="Type d'artiste")

class Show(models.Model):
    # Clé primaire
    id = models.BigAutoField(primary_key=True, verbose_name="ID du spectacle")
    # Clé étrangère vers Location
    location = models.ForeignKey('Location', on_delete=models.SET_NULL, null=True, verbose_name="Lieu")
    slug = models.SlugField(max_length=60, unique=True, verbose_name="Slug")
    title = models.CharField(max_length=255, verbose_name="Titre du spectacle")
    poster_url = models.CharField(max_length=255, null=True, verbose_name="URL de l'affiche")
    bookable = models.BooleanField(verbose_name="Réservable")
    price = models.DecimalField(max_digits=10, decimal_places=2, verbose_name="Prix du spectacle")
    description = models.TextField(verbose_name="Description du spectacle")
    created_at = models.DateField(null=True, verbose_name="Date de création")

class Locality(models.Model):
    # Clé primaire
    id = models.BigAutoField(primary_key=True, verbose_name="ID de la localité")
    postal_code = models.CharField(max_length=6, unique=True, verbose_name="Code postal")
    locality = models.CharField(max_length=60, unique=True, verbose_name="Localité")

class Location(models.Model):
    # Clé primaire
    id = models.BigAutoField(primary_key=True, verbose_name="ID du lieu")
    # Clé étrangère vers Locality
    locality = models.ForeignKey(Locality, on_delete=models.SET_NULL, null=True, verbose_name="Localité")
    slug = models.SlugField(max_length=60, unique=True, verbose_name="Slug")
    designation = models.CharField(max_length=60, verbose_name="Désignation du lieu")
    address = models.CharField(max_length=255, null=True, verbose_name="Adresse")
    website = models.CharField(max_length=255, null=True, verbose_name="Site Web")
    phone = models.CharField(max_length=30, null=True, verbose_name="Numéro de téléphone")

class Collaboration(models.Model):
    # Clé primaire
    id = models.BigAutoField(primary_key=True, verbose_name="ID de la collaboration")
    # Clé étrangère vers ArtistType
    artist_type = models.ForeignKey(ArtistType, on_delete=models.CASCADE, verbose_name="Artiste et son type")
    # Clé étrangère vers Show
    show = models.ForeignKey(Show, on_delete=models.CASCADE, verbose_name="Spectacle")
